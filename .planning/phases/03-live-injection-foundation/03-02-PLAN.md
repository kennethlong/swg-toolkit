struct EngineHookPoint  { const char* name; void* addr; };
struct EngineHookPoints { unsigned version; unsigned count; const EngineHookPoint* entries; };
#define ENGINE_HOOKPOINTS_VERSION 6  // currently 99 names; advisory — contract is name-keyed

<!-- resolveFromExe pattern — port from endpoints_bindings.cpp:802-825 -->
HMODULE hExe = GetModuleHandleA(nullptr);   // GetModuleHandle(NULL) = the host EXE (export lives in the exe)
auto pGet = (const EngineHookPoints*(__cdecl*)())GetProcAddress(hExe, "GetEngineHookPoints");
if (!pGet) { s_advertisedClient = false; return false; }  // SWGEmu legacy: RVA literals stand, strict no-op
s_advertisedClient = true;
const EngineHookPoints* table = pGet();     // CALL it — NEVER scrape the static array (static-init race)
resolve(table, s_bindings, count);          // name-keyed overwrite of fn-pointer slots

<!-- lookupByName — Win32-free (PATTERNS §resolve.cpp endpoints.cpp:169-175) -->
// A missing name leaves the slot at its RVA literal (never nulls — graceful)
static void* lookupByName(const EngineHookPoints* table, const char* name) {
    for (unsigned i = 0; i < table->count; ++i)
        if (strcmp(table->entries[i].name, name) == 0) return table->entries[i].addr;
    return nullptr;
}
