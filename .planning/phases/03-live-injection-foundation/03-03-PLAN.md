bool channelOpen(const char* mappingName, size_t byteSize);
void channelWrite(const LiveState* state);
void channelClose();

// LiveState struct mirrors LIVE_CHANNEL_LAYOUT from contracts/live-inject.ts
struct LiveState {
    LONG     seqCounter;          // offset 0 (seqlock — managed by channelWrite, not caller)
    float    transform[3][4];     // offset 4, 48 bytes
    uint64_t networkId;           // offset 52
    char     templateName[256];   // offset 60
    uint32_t liveness;            // offset 316: bit0=playerNonNull, bit1=isOver
    // TOTAL = 320 bytes matching LIVE_CHANNEL_LAYOUT.TOTAL_SIZE.length
};
static_assert(sizeof(LiveState) == 320, "LiveState must match LIVE_CHANNEL_LAYOUT.TOTAL_SIZE");
static_assert(offsetof(LiveState, transform) == 4, "transform offset must be 4");
static_assert(offsetof(LiveState, networkId) == 52, "networkId offset must be 52");
static_assert(offsetof(LiveState, templateName) == 60, "templateName offset must be 60");
static_assert(offsetof(LiveState, liveness) == 316, "liveness offset must be 316");

<!-- Seqlock pattern from PATTERNS.md §channel.cpp — RESEARCH §Architecture Patterns -->
// Seqlock write: increment seq (odd = write in progress), memcpy payload, increment (even = complete)
// Reader on host: read seq, read payload, read seq again; retry if odd or mismatch

    agent/channel.cpp: Implement per the interface block in the context section.

    channelOpen(const char* mappingName, size_t byteSize):
    - s_hMap = OpenFileMappingA(FILE_MAP_WRITE, FALSE, mappingName)
    - If null: return false (host hasn't created the mapping yet — normal during early startup)
    - s_view = MapViewOfFile(s_hMap, FILE_MAP_WRITE, 0, 0, byteSize)
    - If null: CloseHandle(s_hMap); return false
    - Zero-initialize the view on first open: memset(s_view, 0, byteSize)
    - return true

    channelWrite(const LiveState* state):
    - volatile LONG* seq = static_cast<volatile LONG*>(s_view)
    - InterlockedIncrement(seq)   // seq now odd: write in progress
    - memcpy(static_cast<char*>(s_view) + sizeof(LONG), &state->transform, sizeof(LiveState) - sizeof(LONG))
      // Note: copy everything AFTER the seqlock LONG (transform, networkId, templateName, liveness)
    - InterlockedIncrement(seq)   // seq now even: write complete
    // Reader: retry if seq is odd or changes between reads

    channelClose():
    - If s_view: UnmapViewOfFile(s_view); s_view = nullptr
    - If s_hMap: CloseHandle(s_hMap); s_hMap = nullptr

    The s_hMap and s_view are file-static (module-global) handles — the agent has one channel instance.
