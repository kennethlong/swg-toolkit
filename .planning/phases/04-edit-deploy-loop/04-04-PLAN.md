    Export function getManifestPath(studioDir: string): string — returns path.join(studioDir, 'changesets', 'manifest.json').

    Export function readManifest(studioDir: string): WorkspaceChangesetManifest:
    const p = getManifestPath(studioDir). If !fs.existsSync(p): return { activeVersionIndex: -1, changesets: [] }. Return JSON.parse(fs.readFileSync(p, 'utf8')) as WorkspaceChangesetManifest.

    Export function writeManifest(studioDir: string, manifest: WorkspaceChangesetManifest): void:
    const p = getManifestPath(studioDir); const tmp = p + '.tmp'. fs.mkdirSync(path.dirname(p), {recursive: true}). fs.writeFileSync(tmp, JSON.stringify(manifest, null, 2), 'utf8'). fs.renameSync(tmp, p). (Atomic — matches the useLiveService pattern.)

    Export interface SealLayerParams { sealedBy: 'manual' | 'pack'; deltas: StagingEntry[]; label: string; deployRecord?: import('@swg/contracts').CfgDeployRecord }.

    Export async function sealLayer(params: SealLayerParams): Promise<void>:
    useChangesetStore.getState().beginSeal(). const studioDir = useWorkspaceStore.getState().studioDir; if (!studioDir) throw new Error('No workspace open'). const manifest = readManifest(studioDir). const nextIndex = manifest.changesets.length. const cs: SwgChangeset = { id: crypto.randomUUID(), versionIndex: nextIndex, label: params.label || 'changeset ' + (nextIndex + 1), timestamp: new Date().toISOString(), sealedBy: params.sealedBy, deltas: params.deltas, deployRecord: params.deployRecord }. manifest.changesets.push(cs). manifest.activeVersionIndex = nextIndex. try { writeManifest(studioDir, manifest); useChangesetStore.getState().sealComplete(manifest); } catch(e) { useChangesetStore.getState().sealError(String((e as Error)?.message ?? e)); throw e; }.

    Export function setActiveVersion(index: number): void:
    const studioDir = useWorkspaceStore.getState().studioDir; if (!studioDir) throw new Error('No workspace open'). const manifest = readManifest(studioDir). if (index < -1 || index >= manifest.changesets.length) throw new Error('Index out of bounds'). manifest.activeVersionIndex = index. writeManifest(studioDir, manifest). useChangesetStore.getState().setActiveVersion(index). (Non-destructive: only activeVersionIndex changes; changesets array is never mutated — D-04-08.)

    CRITICAL: NO fs.rmSync, fs.rmdirSync, remove_all, PurgeChangesetLayer, snapshot.tar.gz anywhere in this file. Only reads, writes, and the activeVersionIndex pointer change.
  </action>

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|

| T-04-14 | Repudiation | changeset label used later in git commit | mitigate | Label is sanitized before passing to execFile git commit: strip control characters, cap at 256 chars (done in gitLfsService Task in Plan 04-05 — cfgActivator here only stores the label in JSON) |
| T-04-15 | Tampering | setActiveVersion with out-of-bounds index | mitigate | Validate index in range [-1, changesets.length - 1] before writing; throw 'Index out of bounds' |
| T-04-SC | Tampering | npm/pip installs | mitigate | No new npm packages; slopcheck not required |
