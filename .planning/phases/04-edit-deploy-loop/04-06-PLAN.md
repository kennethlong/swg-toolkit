**As a** SWG mod developer, **I want to** stage edited files in a project workspace and build a deployable `.tre` patch that activates via the client config, **so that** I can iterate on mod changes in-game, roll back to any prior state, and version my work safely via Git/LFS.

    async function handleDeploy():
    // W9: mutual exclusion
    if (deployingRef.current) return;
    deployingRef.current = true;
    try {
      const studioDir = useWorkspaceStore.getState().studioDir!;
      const workspaceName = useWorkspaceStore.getState().workspaceName!;

      // W2: Read from version graph, NOT staging store
      let manifest = readManifest(studioDir);
      const stagingEntries = useStagingStore.getState().entries;

      // Auto-seal if there are uncommitted staging entries
      if (stagingEntries.length > 0) {
        setPhase({kind:'building'});  // building includes sealing
        await sealVersion({sealedBy:'pack', entries: stagingEntries, label:'auto (pack)'});
        manifest = readManifest(studioDir);  // re-read — activeVersionId may have changed
      }

      // Deploy from the sealed version
      const flattenedEntries = flatten(manifest.activeVersionId, manifest, studioDir);
      if (flattenedEntries.length === 0) {
        setPhase({kind:'error', step:'build', message:'No entries in version to deploy (flatten returned empty — version is empty)', cfgRestored:false});
        return;
      }

      const patchName = buildPatchName(workspaceName);  // B6+N2: sanitized + UUID fragment
      const outputPath = path.join(studioDir, 'build', patchName);

      setPhase({kind:'building'});
      try { packPatch(flattenedEntries, outputPath); }
      catch (e) { setPhase({kind:'error', step:'build', message:(e as Error).message ?? String(e), cfgRestored:false}); return; }

      setPhase({kind:'activating'});

      if (deployModel === 'shadow-base') {
        // Shadow-base path (04-06b)
        try {
          const shadowRecord = await deployShadowBase(selectedClient!, studioDir, outputPath, (_pct) => {});
          deployRecordRef.current = shadowRecord;
          setDeployedVersion(manifest.activeVersionId!);  // W2: persist deployedVersionId
          setPhase({kind:'done', slot:'shadow-base', cfgPath: shadowRecord.cfgPath});
        } catch (e) {
          setPhase({kind:'error', step:'activate', message:(e as Error).message ?? String(e), cfgRestored:false});
        }
        return;
      }

      // Patch-prepend path
      const clientLiveDir = path.join(selectedClient!.installPath, 'Live');  // 'Live' hardcoded — Windows-only
      try { fs.copyFileSync(outputPath, path.join(clientLiveDir, patchName)); }
      catch (e) { setPhase({kind:'error', step:'activate', message:(e as Error).message ?? String(e), cfgRestored:false}); return; }

      const cfgDir = path.dirname(selectedClient!.cfgRootPath);
      const swgtoolkitCfgPath = path.join(cfgDir, 'swgtoolkit.cfg');
      if (!fs.existsSync(swgtoolkitCfgPath)) {
        fs.writeFileSync(swgtoolkitCfgPath, '[SharedFile]\r\n', {encoding:'utf8'});
      }

      let record: import('@swg/contracts').CfgInsertionRecord | undefined;

    } finally {
      deployingRef.current = false;  // W9: release mutex
    }
