packages/contracts/src/ipc.ts exports:
  HelloRequest     = { type: 'hello'; id: number }                             // id: correlation id required by Plan 03's demux
  HelloResponse    = { type: 'pong'; id: number; value: string }               // id echoes the request
  CrossWriteReq    = { type: 'cross-write'; id: number; value?: number }       // value is a DECOY/optional — the utility re-reads view[1], it does NOT echo this arg (nonce hardening, review fix MEDIUM-4)
  SabReadyMsg      = { type: 'sab-ready'; sab: SharedArrayBuffer }
  SabCrossWriteAck = { type: 'sab-cross-write-ack'; id: number; value: number } // utility re-reads view[1] (the slot the renderer wrote) and returns it — NOT an echo of CrossWriteReq.value
  PortInitMsg      = { type: 'init-port' }
  IpcMessage       = HelloRequest | HelloResponse | CrossWriteReq | SabReadyMsg | SabCrossWriteAck | PortInitMsg

  // CORRELATION ID (review fix MEDIUM / Codex): hello/pong/cross-write/sab-cross-write-ack ALL carry a
  // numeric `id`. Plan 03's main-process demux resolves the matching pending Promise by this id. Without
  // it the demux would access an untyped `data.id` and tsc --noEmit would not be honest.

  } as const
